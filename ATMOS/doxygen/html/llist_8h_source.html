19 /*
20  * instructions for use:
21  *
22  * for example, you have a structure such as
23  *
24  * struct foo {
25  * int i1;
26  * int i2;
27  * }
28  *
29  * that you want to use in a linked list format.
30  *
31  * FIRST, redefine it as
32  *
33  * struct foo {
34  * LL_PTRS;
35  * int i1;
36  * int i2;
37  * }
38  *
39  * this puts the pointer declarations into the struct
40  * in the correct order & with the correct name, etc.
41  *
42  * SECOND, declare the following items:
43  *
44  * llobject_t ANCHOR;
45  * struct foo *listptr = (struct foo *)&ANCHOR;
46  *
47  * this gives you the head of the list; the way the package
48  * works is you waste a linked list object as the head, but
49  * it's the only way to know you've hit the end (or beginning)
50  * of the list. it's obviously a better use of space to have
51  * ANCHOR be a llobject_t than a struct foo, also.
52  *
53  * THIRD, you need to LL_INIT the list before you use it,
54  * otherwise your program will core dump. note that
55  * whenever you do operations on the list, you should
56  * refer to it as listptr, not ANCHOR or &ANCHOR.
57  *
58  * LL_INIT(listptr);
59  *
60  * this is the right format.
61  *
62  * FOURTH, use the macros and not the pointers or functions
63  * directly - there is a bit of null pointer checking that
64  * goes on when you use the macros, and they are designed to
65  * hide the ANCHOR from you ... you shouldn't have much of
66  * a reason to need to see the ANCHOR, and if you want to
67  * turn off the pointer checking, just set LL_DEBUG to 0.
68  *
69  * function descriptions:
70  *
71  * llobject_t *
72  * LL_NEXT(listptr, elementp)
73  * returns the following item in the list; if the item in
74  * question is the last in list, returns the first item in list
75  *
76  * llobject_t *
77  * LL_PREV(listptr, elementp)
78  * returns the previous item in the list; if the item in
79  * question is the first in list, returns the last item in list
80  *
81  * LL_EACH(listptr, elementp, pointer-type)
82  * iterates through the list, starting with the first & ending
83  * with the last item. to be used like this:
84  * struct foo *tmp;
85  * LL_EACH (listptr, tmp, pointer-type) {
86  * do something with tmp
87  * }
88  *
89  * LL_INIT(listptr)
90  * to be called before ANYTHING is done with a given list structure
91  *
92  * int
93  * LL_IS_EMPTY(listptr)
94  * boolean: returns whether or not the list has 0 items in it
95  *
96  * llobject_t *
97  * LL_R_INSERT(insertpoint, elementp)
98  * inserts the element AFTER the insert point element, returns element
99  *
100  * llobject_t *
101  * LL_L_INSERT(insertpoint, elementp)
102  * inserts the element BEFORE the insert point element, returns element
103  *
104  * llobject_t *
105  * LL_PUSH(listptr, elementp)
106  * inserts the element at the HEAD of the list, returns element
107  *
108  * llobject_t *
109  * LL_APPEND(listptr, elementp)
110  * inserts the element at the END of the list, returns element
111  *
112  * llobject_t *
113  * LL_DETACH(listptr, elementp)
114  * removes the element from the list, returns element
115  *
116  * llobject_t *
117  * LL_POP(listptr)
118  * removes the TOP element from the list,
119  * returns element or NULL if list empty
120  *
121  * llobject_t *
122  * LL_ASSERT(elementp)
123  * prints out an informative message if elementp is not null.
124  * returns element.
125  *
126  * llobject_t *
127  * LL_FIRST(listptr)
128  * LL_HEAD(listptr)
129  * LL_TOP(listptr)
130  * returns first element in list if nonempty. otherwise returns NULL

133  * LL_LAST(listptr)
134  * LL_BOTTOM(listptr)
135  * LL_TAIL(listptr)
136  * returns last element in list if nonempty. otherwise returns NULL
137  */
